SELECT 
    'Table: ' || table_name || E'\n  - ' || 
    string_agg(column_name || ' (' || data_type || ')', E'\n  - ' ORDER BY ordinal_position) AS schema_doc
FROM 
    information_schema.columns
WHERE 
    table_schema = 'public'
GROUP BY 
    table_name
ORDER BY 
    table_name;
